    document.addEventListener('DOMContentLoaded', () => {

      const progressBar = document.querySelector('.progress-bar');

      const progressContainer = document.querySelector('.progress-container');

      const currentTimeEl = document.querySelector('.current-time');

      const durationEl = document.querySelector('.duration');

      const volumeSlider = document.querySelector('.volume-slider');

      const volumeIcon = document.querySelector('.volume-icon');

     

      // 创建音频对象

      const audio = new Audio();

      // 设置音频源 - 使用本地文件路径

      audio.src = 'guangliang.mp3'; // 确保该文件与HTML文件在同一目录下

      audio.preload = 'metadata';

      audio.volume = 0.7;

 

      // 播放状态

      let isPlaying = false;

     

      // 格式化时间

      const formatTime = (seconds) => {

        const mins = Math.floor(seconds / 60);

        const secs = Math.floor(seconds % 60);

        return `${mins}:${secs < 10 ? '0' : ''}${secs}`;

      };

     

      // 更新进度条

      const updateProgress = () => {

        const percent = (audio.currentTime / audio.duration) * 100;

        progressBar.style.width = `${percent}%`;

        currentTimeEl.textContent = formatTime(audio.currentTime);

      };

     

      // 设置进度

      const setProgress = (e) => {

        const width = progressContainer.clientWidth;

        const clickX = e.offsetX;

        const percent = (clickX / width) * 100;

        progressBar.style.width = `${percent}%`;

        audio.currentTime = (percent / 100) * audio.duration;

        currentTimeEl.textContent = formatTime(audio.currentTime);

      };

     

      // 播放/暂停功能

      const togglePlay = () => {

        if (isPlaying) {

          audio.pause();

          vinylDisc.style.animationPlayState = 'paused';

          vinylArm.style.transform = 'rotate(-30deg)';

          playBtn.innerHTML = '<i class="fas fa-play"></i> 播放《光亮》';

          playBtn.classList.remove('playing');

        } else {

          audio.play()

            .catch(error => {

              console.error('播放失败:', error);

              // 显示错误提示

              alert('播放失败: 请确保音频文件存在且路径正确');

            });

          vinylDisc.style.animationPlayState = 'running';

          vinylArm.style.transform = 'rotate(-5deg)';

          playBtn.innerHTML = '<i class="fas fa-pause"></i> 暂停播放';

          playBtn.classList.add('playing');

        }

        isPlaying = !isPlaying;

      };

     

      // 音量控制

      const setVolume = () => {

        audio.volume = volumeSlider.value;

        volumeIcon.textContent = audio.volume === 0 ? '🔇' : '';

      };

     

      // 音频加载后设置总时长

      audio.addEventListener('loadedmetadata', () => {

        durationEl.textContent = formatTime(audio.duration);

      });

     

      // 音频播放时更新进度条

      audio.addEventListener('timeupdate', updateProgress);

     

      // 音频结束时重置播放器

      audio.addEventListener('ended', () => {

        isPlaying = false;

        vinylDisc.style.animationPlayState = 'paused';

        vinylArm.style.transform = 'rotate(-30deg)';

        playBtn.innerHTML = '<i class="fas fa-play"></i> 播放《光亮》';

        playBtn.classList.remove('playing');

        progressBar.style.width = '0%';

        currentTimeEl.textContent = '0:00';

        audio.currentTime = 0;

      });

     

      // 事件监听

      playBtn.addEventListener('click', togglePlay);

      progressContainer.addEventListener('click', setProgress);

      volumeSlider.addEventListener('input', setVolume);
